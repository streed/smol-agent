/**
 * Standardized error class for tools.
 * Provides consistent error structure with codes and details.
 *
 * @example
 * throw new ToolError(ToolErrorCode.FILE_NOT_FOUND, 'File not found', { path: '/test/file.js' });
 *
 * @example
 * const error = ToolError.fileNotFound('/test/file.js');
 * throw error;
 */
export class ToolError extends Error {
  /**
   * Create a ToolError.
   * @param {string} code - Error code from ToolErrorCode
   * @param {string} message - Human-readable error message
   * @param {Object|null} details - Additional error details
   */
  constructor(code, message, details = null) {
    super(message);
    this.name = 'ToolError';
    this.code = code;
    this.details = details;
  }
 
  /**
   * Get human-readable description of error code.
   * @returns {string}
   */
  getDescription() {
    return ErrorCodeDescriptions[this.code] || 'Unknown error';
  }
 
  /**
   * Convert to JSON-serializable object.
   * @returns {Object}
   */
  toJSON() {
    return {
      name: this.name,
      code: this.code,
      message: this.message,
      description: this.getDescription(),
      details: this.details
    };
  }
 
  // ============ Factory Methods ============
 
  /**
   * Create FILE_NOT_FOUND error.
   * @param {string} path - File path
   * @param {Object} [details] - Additional details
   * @returns {ToolError}
   */
  static fileNotFound(path, details = null) {
    return new ToolError(
      ToolErrorCode.FILE_NOT_FOUND,
      `File not found: ${path}`,
      { path, ...details }
    );
  }
 
  /**
   * Create PATH_NOT_ALLOWED error.
   * @param {string} path - Path that was rejected
   * @param {string} reason - Why path is not allowed
   * @param {Object} [details] - Additional details
   * @returns {ToolError}
   */
  static pathNotAllowed(path, reason, details = null) {
    return new ToolError(
      ToolErrorCode.PATH_NOT_ALLOWED,
      reason,
      { path, reason, ...details }
    );
  }
 
  /**
   * Create PERMISSION_DENIED error.
   * @param {string} resource - Resource that was denied
   * @param {string} reason - Why permission was denied
   * @param {Object} [details] - Additional details

    return new ToolError(

  }
 
  /**

   * @returns {ToolError}
   */
  static validationFailed(errors, details = null) {
    return new ToolError(
      ToolErrorCode.VALIDATION_FAILED,
      `Validation failed: ${errors.join(', ')}`,
      { errors, ...details }
    );
  }
 
  /**
   * Create TIMEOUT error.
   * @param {string} operation - What timed out
   * @param {number} timeout - Timeout in milliseconds
   * @param {Object} [details] - Additional details

  static timeout(operation, timeout, details = null) {
    return new ToolError(
      ToolErrorCode.TIMEOUT,
      `Operation timed out after ${timeout}ms: ${operation}`,
      { operation, timeout, ...details }
    );
  }

    return new ToolError(
      ToolErrorCode.EXECUTION_FAILED,
