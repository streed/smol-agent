async function readSnippet(cwd: string, filename: string, maxLines: number): Promise<string | null> {
  try {
    const raw = await fs.readFile(path.join(cwd, filename), "utf-8");
    return raw.split("\n").slice(0, maxLines).join("\n");
  } catch {
    return null;
  }
}
